    const match = url.match(/(?:youtu\.be\/|youtube\.com(?:\/embed\/|\/v\/|\/watch\?v=|\/watch\?.+&v=))([\w-]{11})/);
    return match ? match[1] : null;
};

                                    )}
                                    {item.category && (
                                        <div className="text-gray-600 text-sm mb-2">{item.category}</div>
                                    )}
                                </div>
                            </div>
                        ))}
                    </div>
                </article>
            );
